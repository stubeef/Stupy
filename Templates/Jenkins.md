## Create text files via command line

```
echo 'DROP TABLE SHARED_ANALYSIS.MARKETING.MKTG_MA_VENDOR_UPDATE IF EXISTS'
 > 01_MKTG_MA_VENDOR_DROP.txt
echo 'DROP TABLE SHARED_ANALYSIS.MARKETING.MKTG_MA_VENDOR_OLD IF EXISTS'

 > 03_MKTG_MA_VENDOR_DROP_OLD.txt
 
echo 'ALTER TABLE SHARED_ANALYSIS.MARKETING.MKTG_MA_VENDOR RENAME TO SHARED_ANALYSIS.MARKETING.MKTG_MA_VENDOR_OLD'
 > 04_MKTG_MA_VENDOR_RENAME_OLD.txt
 
echo 'ALTER TABLE SHARED_ANALYSIS.MARKETING.MKTG_MA_VENDOR_UPDATE RENAME TO SHARED_ANALYSIS.MARKETING.MKTG_MA_VENDOR' 
> 05_MKTG_MA_VENDOR_RENAME_UPDATE.txt

echo 'GENERATE STATISTICS ON SHARED_ANALYSIS.MARKETING.MKTG_MA_VENDOR' 
> 06_MKTG_MA_VENDOR_GEN_STATS.txt
```
